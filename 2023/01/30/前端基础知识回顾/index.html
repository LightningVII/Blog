    colorList.splice(2, 1) –> [“orange”]; colorList –> [‘red’, ‘green’]

    nodeList.slice(1) –> Uncaught TypeError: nodeList.slice is not a function
    Array.prototype.slice.call(nodeList, 1) –> [li, li]
